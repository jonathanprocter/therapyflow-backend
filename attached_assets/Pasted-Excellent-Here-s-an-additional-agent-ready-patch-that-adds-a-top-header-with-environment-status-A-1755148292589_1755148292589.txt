+function useApiHealth(pollMs = 10000) {
+  const [healthy, setHealthy] = React.useState(null);
+  const [version, setVersion] = React.useState("–");
+  const [lastChecked, setLastChecked] = React.useState("–");
+
+  const check = React.useCallback(async () => {
+    try {
+      const r = await fetch("/api/health", { method: "GET" });
+      if (!r.ok) throw new Error(`health ${r.status}`);
+      const j = await r.json();
+      setHealthy(!!j.ok);
+      setVersion(j.version || "dev");
+      setLastChecked(new Date().toLocaleTimeString());
+    } catch {
+      setHealthy(false);
+      setLastChecked(new Date().toLocaleTimeString());
+    }
+  }, []);
+
+  React.useEffect(() => {
+    check();
+    const t = setInterval(check, pollMs);
+    return () => clearInterval(t);
+  }, [check, pollMs]);
+
+  return { healthy, version, lastChecked, check };
+}

Optional environment wiring
- If you want the ENV label to reflect a build-time variable, add VITE_APP_ENV to your client environment (e.g., development, staging, production). In Replit/Vite, you can set this in your .env or Replit Secrets as VITE_APP_ENV.
- The APP_VERSION used by /api/health can be set via server-side environment variable APP_VERSION, or you can compute from your git SHA if desired.
