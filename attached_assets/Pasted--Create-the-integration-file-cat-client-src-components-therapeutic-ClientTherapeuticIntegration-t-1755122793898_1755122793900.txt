/**
 * OPTION 3: Progress Note Enhancement
 * Adds therapeutic insights to progress note view
 */
export function EnhancedProgressNoteView({ 
  noteId, 
  clientId 
}: { 
  noteId: string; 
  clientId: string;
}) {
  const [note, setNote] = useState<any>(null);
  const [tags, setTags] = useState<any[]>([]);
  const [insights, setInsights] = useState<any[]>([]);

  useEffect(() => {
    // Fetch note and related therapeutic data
    Promise.all([
      fetch(`/api/progress-notes/${noteId}`).then(r => r.json()),
      fetch(`/api/therapeutic/tags/${clientId}?sessionId=${noteId}`).then(r => r.json()),
      fetch(`/api/therapeutic/insights/${clientId}?sessionId=${noteId}`).then(r => r.json())
    ]).then(([noteData, tagsData, insightsData]) => {
      setNote(noteData.note);
      setTags(tagsData.tags || []);
      setInsights(insightsData.insights || []);
    });
  }, [noteId, clientId]);

  if (!note) return <div>Loading...</div>;

  return (
    <div className="space-y-6">
      {/* Original Note Content */}
      <Card>
        <CardHeader>
          <CardTitle>Progress Note</CardTitle>
          <p className="text-sm text-muted-foreground">
            {new Date(note.sessionDate).toLocaleDateString()}
          </p>
        </CardHeader>
        <CardContent>
          <p className="whitespace-pre-wrap">{note.content}</p>
        </CardContent>
      </Card>

      {/* AI-Extracted Information */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        <Card>
          <CardHeader>
            <CardTitle className="text-sm">Extracted Themes & Emotions</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="space-y-2">
              {tags.map((tag, idx) => (
                <div key={idx} className="flex flex-wrap gap-1">
                  <Badge variant="outline" className="text-xs">
                    {tag.category}
                  </Badge>
                  {tag.tags?.map((t: string, i: number) => (
                    <Badge key={i} variant="secondary" className="text-xs">
                      {t}
                    </Badge>
                  ))}
                </div>
              ))}
            </div>
          </CardContent>
        </Card>

        <Card>
          <CardHeader>
            <CardTitle className="text-sm">Key Insights</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="space-y-2">
              {insights.map((insight, idx) => (
                <div key={idx} className="text-sm p-2 bg-secondary rounded">
                  {insight.insight}
                </div>
              ))}
            </div>
          </CardContent>
        </Card>
      </div>
    </div>
  );
}

/**

function InsightsPreview({ clientId, limit = 3 }: { clientId: string; limit?: number }) {
  const [insights, setInsights] = useState<any[]>([]);

  useEffect(() => {
    fetch(`/api/therapeutic/insights/${clientId}?limit=${limit}`)
      .then(res => res.json())
      .then(data => setInsights(data.insights || []))
      .catch(console.error);
  }, [clientId, limit]);

  if (insights.length === 0) {
    return (
      <p className="text-sm text-muted-foreground">
        No insights yet. They'll appear as sessions progress.
      </p>
    );
  }

  return (
    <div className="space-y-2">
      {insights.map((insight) => (
        <div key={insight.id} className="flex items-start gap-2">
          <div className="w-2 h-2 rounded-full bg-purple-500 mt-1.5" />
          <p className="text-sm">{insight.insight}</p>
        </div>
      ))}
    </div>
  );
}

/**
 * Custom hook for accessing therapeutic data
 */

import { useState, useEffect } from 'react';

interface TherapeuticData {
  insights: any[];
  tags: any[];
  synthesis: any;
  loading: boolean;
  error: Error | null;
}

export function useTherapeuticData(clientId: string): TherapeuticData {
  const [data, setData] = useState<TherapeuticData>({
    insights: [],
    tags: [],
    synthesis: null,
    loading: true,
    error: null
  });

  useEffect(() => {
    if (!clientId) return;

    const fetchData = async () => {
      try {
        const [insightsRes, tagsRes] = await Promise.all([
          fetch(`/api/therapeutic/insights/${clientId}`),
          fetch(`/api/therapeutic/tags/${clientId}`)
        ]);

        const [insightsData, tagsData] = await Promise.all([
          insightsRes.json(),
          tagsRes.json()
        ]);

        setData({
          insights: insightsData.insights || [],
          tags: tagsData.tags || [],
          synthesis: null,
          loading: false,
          error: null
        });
      } catch (error) {
        setData(prev => ({
          ...prev,
          loading: false,
          error: error as Error
        }));
      }
    };

    fetchData();
  }, [clientId]);

  return data;
}

export function useTherapeuticInsightCount(clientId: string): number {
  const [count, setCount] = useState(0);

  useEffect(() => {
    if (!clientId) return;

    fetch(`/api/therapeutic/insights/${clientId}/count`)
      .then(res => res.json())
      .then(data => setCount(data.count || 0))
      .catch(() => setCount(0));
  }, [clientId]);

  return count;
}
